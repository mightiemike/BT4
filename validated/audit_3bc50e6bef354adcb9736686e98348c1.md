### Summary
A malicious IBC counterparty can send a `NonFungibleTokenPacketData` packet whose `ClassId` already carries the receiving chain's destination prefix (e.g., `"nft/channel-0/nftClass"`). The original escrowed NFT can never be correctly recovered.

### Finding Description

func ParseClassTrace(rawClassID string) ClassTrace {
	classSplit := strings.Split(rawClassID, "/")

	if classSplit[0] == rawClassID {
		return ClassTrace{
			Path:        "",
			BaseClassId: rawClassID,
		}
	}

	return ClassTrace{
		Path:        strings.Join(classSplit[:len(classSplit)-1], "/"),
		BaseClassId: classSplit[len(classSplit)-1],
	}
}
